409      /** 
410       * Initialize and Start the Camera based on the CamIdz 
411       * @param CamIdx 
412       * @param orientation 
413       * @param rotation 
414       * @return 
415       */ 
416      public long initializeCamera(int CamIdx,int orientation, int rotation) { 
417   
418   
419          //setContentView(R.layout.activity_camera); 
420          if (ContextCompat.checkSelfPermission(this, Manifest.permission.CAMERA) != PackageManager.PERMISSION_GRANTED) { 
421   
422              ActivityCompat.requestPermissions(this, new String[] { Manifest.permission.CAMERA, Manifest.permission.WRITE_EXTERNAL_STORAGE }, 0); 
423          } 
424          while(ContextCompat.checkSelfPermission(this, Manifest.permission.CAMERA)  != PackageManager.PERMISSION_GRANTED) { 
425   
426          } 
427          mCamera[CamIdx] = Camera.open(CamIdx); 
428          mPicture[CamIdx] = getPictureCallback(); 
429          Camera.Size result = null; 
430   
431          try { 
432              Camera.Parameters param = mCamera[CamIdx].getParameters(); 
433              List sizes = param.getSupportedPictureSizes(); 
434              focus_mode = param.getSupportedFocusModes(); 
435              result = (Size) sizes.get(0); 
436              param.setRotation(rotation); 
437              param.setPreviewSize(320,240); 
438              String str = mCamera[CamIdx].getParameters().flatten(); 
439              Log.e("DualCam", str); 
440              ev_max = param.getMaxExposureCompensation(); 
441              ev_min = param.getMinExposureCompensation(); 
442              param.setAntibanding("auto"); 
443              param.setPictureSize(result.width,result.height); 
444              if(focus_mode.contains("FOCUS_MODE_CONTINUOUS_PICTURE")) //TODO: changes here done 
445                  param.setFocusMode(FOCUS_MODE_CONTINUOUS_PICTURE); 
446              mCamera[CamIdx].setParameters(param); 
447              /**Preview Callback for YUV Dump*/ 
448              int dataBufferSize=(int)(320*240* 
449                      (ImageFormat.getBitsPerPixel(param.getPreviewFormat())/8.0)); 
450   
451              for(int i=0; i<10;i++) { 
452                  callbackBuffer =new byte[dataBufferSize]; 
453                  mCamera[CamIdx].addCallbackBuffer(callbackBuffer); 
454              } 
455              mCamera[CamIdx].setPreviewCallbackWithBuffer(previewCallback); 
456              /*Code for Yuv Dump ends here */ 
457   
458              mCamera[CamIdx].setDisplayOrientation(orientation); 
459              mCamera[CamIdx].setPreviewTexture(Surface[CamIdx]); 
460              mCamera[CamIdx].startPreview(); 
461              } catch (IOException ioe) { 
462              // Something bad happened 
463          } 
464   
465          return 0; 
466      }

807                  break; 
808          } 
809          //CameraModeIdx = CAMERA_MODE_INVALID; 
810          return 0;

814       * Starts the Camera with the provided parameters 
815       * @param CamIdx 
816       * @param Surface 
817       * @param orientation 
818       * @param rotation 
819       * @return 
820       */ 
821      public long startCamera(int CamIdx,SurfaceTexture Surface ,int orientation,int rotation ) { 
822          /*Start Camera*/ 
823   
824          Camera camera = Camera.open(CamIdx); 
825          try { 
826              Camera.Parameters params = camera.getParameters(); 
827              params.setRotation(rotation); 
828   
829              camera.setDisplayOrientation(orientation); 
830              camera.setPreviewTexture(Surface); 
831              if(focus_mode.contains("FOCUS_MODE_CONTINUOUS_PICTURE")) //TODO: changes here done 
832                  params.setFocusMode(FOCUS_MODE_CONTINUOUS_PICTURE); 
833              camera.setParameters(params); 
834   
835          } catch (IOException e) { 
836              e.printStackTrace(); 
837          } 
838          camera.startPreview(); 
839          mCamera[CamIdx] = camera; 
840   
841   
842          return 0; 
843      }
